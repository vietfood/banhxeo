class ModelConfig(BaseModel):
    vocab_size: Optional[int] = None  # might be useful for all models

    def __str__(self):
        params = ", ".join(f"{k}={v}" for k, v in self.model_dump().items())
        return f"{self.__class__.__name__}({params})"
